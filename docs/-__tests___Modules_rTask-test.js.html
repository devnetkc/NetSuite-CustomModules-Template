/** @format
 * @module rTask
 */

const nTask = require('N/task');
const log = require('N/log');
const rRuntime = require('N/runtime');

const rTask = require('../../src/FileCabinet/SuiteScripts/Modules/rTask.js');

jest.mock('N/task');
jest.mock('N/log');

beforeEach(() => {
  jest.clearAllMocks();
});

// eslint-disable-next-line import/no-amd
describe('MOD_cfm_RAV_task.js', () => {
  const NewTask = new nTask.ScheduledScriptTask();
  nTask.create = jest.fn().mockReturnValue(NewTask);
  //jest.spyOn(RuntimeScript,'getRemainingUsage').mockReturnValue(1000)
  describe('Governance should be high enough to return true', () => {
    const RuntimeScript = new rRuntime.Script();
    RuntimeScript.getRemainingUsage = jest.fn().mockReturnValue(1000);
    test('Should be false with run limit of 0', () => {
      const TestGovernanceOptions = {
        curScript: RuntimeScript,
        list: [{ name: 'test list name' }],
        counts: { limitRunsTo: '0', numberOfRecords: 4 },
      };
      const CheckGovernance = rTask.CheckGovernance(TestGovernanceOptions);
      expect(CheckGovernance).toBe(false);
    });
    test('Should be true with no run limit of -1', () => {
      const TestGovernanceOptions = {
        curScript: RuntimeScript,
        list: [{ name: 'test list name' }],
        counts: { limitRunsTo: '-1', numberOfRecords: 4 },
      };
      const CheckGovernance = rTask.CheckGovernance(TestGovernanceOptions);
      expect(CheckGovernance).toBe(true);
    });
  });
  describe('Governance should be low enough to return false', () => {
    const RuntimeScript = new rRuntime.Script();
    RuntimeScript.getRemainingUsage = jest.fn().mockReturnValue(90);
    test('Should be false with run limit of 100', () => {
      const TestGovernanceOptions = {
        curScript: RuntimeScript,
        list: [{ name: 'test list name' }],
        counts: { limitRunsTo: '100', numberOfRecords: 4 },
      };
      const CheckGovernance = rTask.CheckGovernance(TestGovernanceOptions);
      expect(CheckGovernance).toBe(false);
    });
    test('Should be false with no run limit of -1', () => {
      const TestGovernanceOptions = {
        curScript: RuntimeScript,
        list: [{ name: 'test list name' }],
        counts: { limitRunsTo: '-1', numberOfRecords: 4 },
      };
      const CheckGovernance = rTask.CheckGovernance(TestGovernanceOptions);
      expect(CheckGovernance).toBe(false);
    });
  });
});
